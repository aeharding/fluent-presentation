Destructuring


let [first, , third, rest] = [1, 2, 3, 4, 5, 6];

first === 1; // true
third === 3; // true
// rest equals [4, 5, 6]


let person = {name: 'Alex', phone: '+15031234567'};

let {name: name} = person;

Defaults

Default is used if property is strictly undefined


let person = {name: 'Alex', phone: '+15031234567'};

let {email='lolnope'} = person;

email === 'lolnope'; // true

QUIZ!


let person = {name: 'Alex', phone: null};

let {phone='911'} = person;

Q: What does phone equal?

A: null
